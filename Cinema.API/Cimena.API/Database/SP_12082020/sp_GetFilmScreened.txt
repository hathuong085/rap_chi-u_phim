CREATE PROCEDURE [dbo].[sp_GetFilmScreened]
	@CateId int
AS
BEGIN
	Select * from Film f where (f.Status='true' and f.CategoryId=@CateId and
			NOT EXISTS(Select distinct s.FilmId from Showing s where s.FilmId=f.FilmId and s.StatusId=1) )
END
GO
